525 __STATIC_INLINE uint8_t __CLZ(uint32_t data)
526 {
527 if (data == 0U) { return 32U; }
528
529 uint32_t count = 0U;
530 uint32_t mask = 0x80000000U;
531
532 while ((data & mask) == 0U)
533 {
534 count += 1U;
535 mask = mask >> 1U;
536 }
537 return count;
538 }

540 __STATIC_INLINE uint32_t __RBIT(uint32_t v)
541 {
542 uint8_t sc = 31U;
543 uint32_t r = v;
544 for (v >>= 1U; v; v >>= 1U)
545 {
546 r <<= 1U;
547 r |= v & 1U;
548 sc--;
549 }
550 return (r << sc);
551 }
